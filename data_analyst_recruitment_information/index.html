def cut_word(word, method):
    # `-` `7k-8k`
    position = word.find('-')
    length = len(word)
    
    #  `7k-8k`
    if position != -1:
        bottomSalary = word[:position-1]
        topSalary = word[position+1:length-1]
        
    #  `9k/9K`
    else: 
        bottomSalary = word[:word.upper().find('K')]
        topSalary = bottomSalary
        
    if method == 'bottom':
        return bottomSalary
    else:
        return topSalary
